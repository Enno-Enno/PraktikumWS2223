    \intertext{Dies entspricht den Abweichungen}
    \nonumber \frac{|\nu_\text{res} - \nu_\text{res,theo}|}{\nu_\text{res,theo}} &= (\num{154.9} \pm \num{0.4}) \, \%, \\
